/********************************************************************************************
***** The Next sub rutine will recored and will also 
***** show the time to each subrutine will be 
********************************************************************************************/

Sub Exe_sub(sub_routine)
    Trace The routine $(sub_routine) just started;
    Let sub_start = Num(Now(1));
    Call $(sub_routine)
    Let sub_finish = Num(Now(1));
    Let sub_duration = Num(Round((sub_finish-sub_start)*86400),'#,##0');
    Trace The rutine $(sub_routine) completed, Time:$(sub_duration) seconds.;
End Sub


Sub loadUsers
/********************************************
***  SubRutine to load Users MetaData
***
********************************************/
	SET vParams = 'limit=100';
    
  	Do
        RestConnectorMasterTable:
        SQL SELECT 
            "__KEY_root",
            (SELECT 
                "id" AS "id_u2",
                "tenantId",
                "status",
                "subject",
                "name" AS "name_u2",
                "email",
                "createdAt",
                "lastUpdatedAt",
                "created",
                "lastUpdated",
                "__KEY_data",
                "__FK_data",
                (SELECT 
                    "@Value",
                    "__FK_roles"
                FROM "roles" FK "__FK_roles" ArrayValueAlias "@Value"),
                (SELECT 
                    "id",
                    "name",
                    "type",
                    "level",
                    "__FK_assignedRoles"
                FROM "assignedRoles" FK "__FK_assignedRoles"),
                (SELECT 
                    "id" AS "id_u1",
                    "name" AS "name_u1",
                    "providerType",
                    "__KEY_assignedGroups",
                    "__FK_assignedGroups",
                    (SELECT 
                        "id" AS "id_u0",
                        "name" AS "name_u0",
                        "type" AS "type_u0",
                        "level" AS "level_u0",
                        "__FK_assignedRoles_u0"
                    FROM "assignedRoles" FK "__FK_assignedRoles_u0")
                FROM "assignedGroups" PK "__KEY_assignedGroups" FK "__FK_assignedGroups"),
                (SELECT 
                    "__KEY_links",
                    "__FK_links",
                    (SELECT 
                        "href",
                        "__FK_self"
                    FROM "self" FK "__FK_self")
                FROM "links" PK "__KEY_links" FK "__FK_links")
            FROM "data" PK "__KEY_data" FK "__FK_data"),
            (SELECT 
                "__KEY_links_u0",
                "__FK_links_u0",
                (SELECT 
                    "href" AS "href_u0",
                    "__FK_self_u0"
                FROM "self" FK "__FK_self_u0"),
                (SELECT 
                    "href" AS "href_u1",
                    "__FK_next"
                FROM "next" FK "__FK_next")
            FROM "links" PK "__KEY_links_u0" FK "__FK_links_u0")
        FROM JSON (wrap on) "root" PK "__KEY_root"
        WITH CONNECTION ( URL "https://$(vu_tenant_fqdn)/api/v1/users?$(vParams)" );

        [tmp_Usr_Roles]:
        LOAD	
            [@Value] as 'Role',
            [id_u1] as 'Role_id',
            [__FK_roles] as 'Usr_KeyData'
        RESIDENT RestConnectorMasterTable
        WHERE NOT IsNull([__FK_roles]);

        [tmp_Usr_Groups]:
        LOAD	
            [id_u1] AS [id_u1],
            [name_u1] AS 'Group',
            //[providerType] ,
            [__FK_assignedGroups] as 'Usr_KeyData'
        RESIDENT RestConnectorMasterTable
        WHERE NOT IsNull([__FK_assignedGroups]);

        [tmp_Users]:
        LOAD Distinct
            "id_u2" AS 'id',
            "tenantId",
            "status",
            "subject",
            "name_u2"  AS 'name',
            "email" ,
            "createdAt",
            "lastUpdatedAt",
            "created" ,
            "lastUpdated"
            //,[__KEY_data] 
        RESIDENT RestConnectorMasterTable
        WHERE NOT IsNull([__FK_data]);
       
        [NextURL]:
        LOAD	[href_u1] AS [NextURL],
            [__FK_next] AS [__KEY_links_u0]
        RESIDENT RestConnectorMasterTable
        WHERE NOT IsNull([__FK_next]);
        
        Drop Table RestConnectorMasterTable;
        
        trace $(vParams);
        LET vParams = SubField(Peek('NextURL',0,'NextURL'),'?',-1);
        LET vNextURLRows = NoOfRows('NextURL');
        
        trace Parameters $(vParams);
        trace Rows  $(vNextURLRows);
        
        Drop Table NextURL;

    Loop while $(vNextURLRows)>0;  
    
    QUALIFY *; // Qualifies all fields in subsequent loads
   
   	[Users]:
   	LOAD *
    ,ApplyMap('All_Usr', subject, 'Unknown') as License_Type
    RESIDENT tmp_Users;
   
   	[Usr_Roles]:
   	LOAD * RESIDENT tmp_Usr_Roles;
   
   	[Usr_Groups]:
   	LOAD * RESIDENT tmp_Usr_Groups;
   
   	UnQUALIFY *; // UnQualifies all fields in subsequent loads

    [Users_stat]:
   	LOAD 
        COUNT(id) as CountValue,
        MAX(createdAt) as MaxValue,
        MIN(createdAt) as MinValue
    RESIDENT tmp_Users;
    
    Let vTotUsr = TEXT(Peek('CountValue',0,'Users_stat'));
    Let VMinUserCreated = TEXT(Date(Peek('MinValue',0,'Users_stat'),'DD-MMM-YYYY'));
    Let VMaxUserCreated = TEXT(Date(Peek('MaxValue',0,'Users_stat'),'DD-MMM-YYYY'));

     [Map_Users]:
     LOAD
         id as ownerId,
         name as User_Name
     RESIDENT tmp_Users;
    
    Trace 'Dropping Temp Tables';
   	Drop Table tmp_Users;
   	Drop Table tmp_Usr_Roles;
   	Drop Table tmp_Usr_Groups;
    Drop Table Users_stat;

End Sub

Sub loadDIProject
/********************************************
***  SubRutine to load Data Integration Project MetaData
***
********************************************/
    RestConnectorMasterTable:
    SQL SELECT 
        "id",
        "name",
        "description",
        "ownerId",
        "spaceId"
    FROM JSON (wrap on) "root"
    WITH CONNECTION ( URL "https://$(vu_tenant_fqdn)/api/v1/di-Projects" );
	
    QUALIFY *; // Qualifies all fields in subsequent loads\
    
    if ScriptErrorcount > 0 then

        [DIP]:
        LOAD * Inline ['id', 'name', 'description','OwnerID','SpaceID','Space_Name','User_Name'];

		Let vScriptErrors = ScriptErrorCount & '|' & num(ScriptError) & '|' & ScriptError & '|' & ScriptErrorDetails & '|' & ScriptErrorList;
    
		TRACE $(vScriptErrors);    

	Else
        [DIP]:
        LOAD	
            [id],
            [name],
            [description],
            [ownerId],
            [spaceId]
            ,ApplyMap('Map_Spaces', spaceId, 'Unknown') as Space_Name
            ,ApplyMap('Map_Users', ownerId, 'Unknown') as User_Name
        RESIDENT RestConnectorMasterTable;
    end if
	
    UNQUALIFY *; // UnQualifies all fields in subsequent loads
    DROP TABLE RestConnectorMasterTable;

End Sub
